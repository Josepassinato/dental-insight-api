Integration Examples

Basic JavaScript Integration:

// Initialize the SDK DentalSDK.init({ apiKey: 'your-clinic-api-key', baseUrl: 'https://your-dental-app.com' }); // Upload an exam const file = document.getElementById('fileInput').files[0]; const result = await DentalSDK.uploadExam(file, { patientName: 'John Doe', dentist: 'Dr. Smith' }); // Open viewer in iframe mode DentalSDK.openViewer(result.examId, { containerId: 'viewer-container', width: 1200, height: 800, mode: 'iframe' });

Web Component Usage:

<!-- After SDK initialization --> <dental-viewer exam-id="your-exam-id" api-key="your-api-key" base-url="https://your-dental-app.com" width="1200" height="800"> </dental-viewer>

React Integration:

import { useEffect, useRef } from 'react'; function DentalViewerComponent({ examId, apiKey }) { const containerRef = useRef(); useEffect(() => { // Load SDK const script = document.createElement('script'); script.src = '/dental-sdk.js'; script.onload = () => { DentalSDK.init({ apiKey: apiKey, baseUrl: 'https://your-dental-app.com' }); DentalSDK.openViewer(examId, { containerId: containerRef.current.id, mode: 'iframe' }); }; document.head.appendChild(script); return () => { DentalSDK.closeViewer(containerRef.current.id); }; }, [examId, apiKey]); return <div id="dental-viewer" ref={containerRef}></div>; }
